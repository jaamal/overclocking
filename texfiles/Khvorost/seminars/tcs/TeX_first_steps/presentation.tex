\section{TeX}

\begin{frame}
\frametitle{Что такое TeX?}

\begin{block}{Определение}<1->
TeX -- набор команд для верски текста.
\end{block}

\begin{block}{Эволюция}
\begin{itemize}
  \item<2-> 1978 г. \emph{Д. Кнут} Оригинальный пакет \textbf{TeX} (Искусство программирования);
  \item<3-> 1984 г. \emph{Л. Лэмпорт} Пакет \textbf{LaTeX} (ориентированный на удобную верстку публикаций);
  \item<4-> 1985 г. \emph{О. Паташник и Л. Лэмпорт} Пакет \textbf{BibTeX} (ориентированный на управление ссылками);
  \item<5-> 1990 г. \emph{Г. Хэйген} Пакет \textbf{ConTeXt} (ориентированный на верстку книг);
  \item<6-> 1994 г. \emph{Д. Хобби} Пакет \textbf{MetaPost} (ориентированный на создание иллюстраций);
\end{itemize}
\end{block}

\end{frame}

\begin{frame}
\frametitle{С чем мы работаем на самом деле?}

\begin{frame}
\frametitle{Знаете ли Вы какие пакеты поддерживаются Вашей сборкой TeX-a?}

\begin{block}{TeXLive (\url{http://tug.org/texlive/})}
	\begin{itemize}
  		\item Бесплатная сборка пакетов TeX-a;
  		\item Поддерживаются почти все пакеты (\textbf{TeX}, \textbf{LaTeX}, \textbf{BibTeX},
  		\textbf{ConTeXt},\textbf{MetaPost});
  		\item Регулярное обновление пакетов;
  		\item Поддержка большинства ОС (Windows, Unix);	
  	\end{itemize}
\end{block}
\end{frame}
