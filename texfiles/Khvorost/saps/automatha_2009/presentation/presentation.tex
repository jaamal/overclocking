\frame {
    \frametitle{Simple Definitions}
    \begin{block}{Primitive string}
        A string $x$ is {\bf primitive} if $x = u^k$ for some $k$ implies that $k = 1$ and $u = x$.
    \end{block}

    \begin{block}{Pure square and repetition}
        A {\bf pure} square is a square $xx$ where $x$ is primitive. Otherwise, $xx$ is called a {\bf repetition}.
    \end{block}
}

\frame {
    \frametitle{Main Problem}
    \begin{block}{Computing All Pure Squares Problem}
        {\sc Input:} an SLP $S$ that derives some text $S$;
        \bigskip
        {\sc Output:} a data structure (a PS-table) that contains information about all pure squares in $S$ in a
compressed form;
    \end{block}
}

\frame {
    \frametitle{Pure Squares Table}
    \begin{block}{PS-table Properties}
        \begin{itemize}
        \item The size of $PS$-table is equal to $(\lfloor\log |S|\rfloor+1) \times (|\mathbb{S}|+1)$;
        \item The cell $PS(i,j)$ with $i,j>0$ contains information about the family of all pure squares such that
                \renewcommand{\labelenumi}{(\theenumi)}
                \begin{enumerate}
	                \item they touch the cut position of the rule $\mathbb{S}_j$;
	                \item they are contained in the text $S_j$;
	                \item lengths of their roots belong to the interval $[2^{i-1}, 2^i)$.
                \end{enumerate}
        \end{itemize}
    \end{block}
}

\frame {
    \frametitle{Summarize}
    \begin{block}{Features of the algorithm}
        \begin{enumerate}
        \item The algorithm runs on $O(\max(|\mathbb{S}|^5\log|S|, |\mathbb{S}|^3\log^3|\mathbb{S}|\log|S|))$
        time and requires $O(\mathbb{S}^3)$ space;
        \item The algorithm is divided into independent steps in contrast to classical algorithms in this area
        which consecutively accumulate information about required objects. As a result it can be parallelized;
        \item The algorithm is not excluded that the constants hidden in the ``$O$'' notation are actually very
        big;
        \end{enumerate}
    \end{block}
}
