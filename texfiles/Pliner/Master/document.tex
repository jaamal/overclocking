\section{Прямолинейные программы}
\subsection{Определение прямолинейной программы}

\begin{frame}
\frametitle{}

\begin{block}{Определение}
\textbf{Прямолинейная программа} (ПП) $\mathcal{X}$ размера $n$~-- это
последовательность правил вывода
\begin{center}
$\mathcal{X}_1 = expr_1, \mathcal{X}_2 = expr_2,\dots,\mathcal{X}_n = expr_n$,
\end{center}
где $\mathcal{X}_i$~-- это \textbf{переменные}, а $expr_i$~-- это
\textbf{выражения} вида:
\begin{itemize}  
  \item $expr_i$~-- символ из алфавита $\Sigma$ (\textbf{терминальные} правила).
  \item $expr_i = \mathcal{X}_l \cdot \mathcal{X}_r$ ($l,r < i$)
 (\textbf{нетерминальные} правила).
\end{itemize}
\end{block}

\pause
\begin{block}{}
\textbf{Прямолинейная программа}~-- это грамматика, выводящая в точности одно
слово.
\end{block}
\end{frame}

\begin{frame}
\frametitle{ПП $\mathcal{X}$, выводящая строку

\begin{center}
$\mathcal{X}_1 = b$\\
$\mathcal{X}_2 = a$\\

$\mathcal{X}_4 = \mathcal{X}_3 \cdot \mathcal{X}_2$\\

$\mathcal{X}_6 = \mathcal{X}_5 \cdot \mathcal{X}_4$\\
$\mathcal{X}_7 = \mathcal{X}_6 \cdot \mathcal{X}_5$
\end{center}
\end{exampleblock}

\end{frame}

\section{Алгоритм Лифшица}
\subsection{Описание}

\begin{frame}
\frametitle{}
\begin{block}{На вход}
ПП размером $n$ выводящая текст и ПП размером $m$ выводящая шаблон.
\end{block}
\pause
\begin{block}{На выход}
Позиции вхождения шаблона в текст.
\end{block}
\pause
\begin{block}{Алгоритмическая сложность}
Алгоритм работает за $O(n^2 \cdot m)$ по времени и за $O(n \cdot m)$ по памяти.
\end{block}
\end{frame}

\subsection{Замечания}
\begin{frame}
\frametitle{}
\begin{block}{Замечание 1}
Если $T = a^{2^n}, P = a^{2^m}$, то алгоритм Лифшица будет работать за время
$O(n^2m)$ (при условии, что для этих текстов уже построены ПП), в то время как
классический алгоритм КМП~-- за время $O(2^n + 2^m)$.
\end{block}
\pause
\begin{block}{Замечение 2}
Если ПП сбалансированны, то сложность работы алгоритма $O(n \cdot log(n) \cdot m)$.
\end{block}
\end{frame}
