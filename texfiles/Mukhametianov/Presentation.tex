\section{Dependency Inversion}

\begin{frame}[fragile]
	\pause
	\begin{block}{Код без соблюдения принципа инверсии зависимостей}
		\begin{lstlisting}[language=Java]
			public A()
			{
			    this.b = new B();
			    this.c = new C();
			}
		\end{lstlisting}
	\end{block}
	\pause
	\begin{block}{Код с соблюдением принципа инверсии зависимостей}
		\begin{lstlisting}[language=Java]
			public A(IB b, IC c)
			{
			    this.b = b;
			    this.c = c;
			}
		\end{lstlisting}
	\end{block}
\end{frame}

\begin{frame}[fragile]
	\begin{block}{Наблюдение}
		Должно быть место, где создаются и хранятся объекты.
	\end{block}
	\pause
	\begin{block}{ServiceLocator}
		\begin{lstlisting}[language=Java]
			ServiceLocator.register(IB.class, new B());
			ServiceLocator.register(IC.class, new C());
			ServiceLocator.register(IA.class, new A(ServiceLocator.get(IB.class), ServiceLocator.get(IC.class)));
		\end{lstlisting}
	\end{block}
\end{frame}

\section{JRoboContainer}

\begin{frame}
	\begin{block}{}
		Можно ли реализовать DI-контейнер, конфигурирование которого не потребует большого количества кода?
	\end{block}
	\pause
	\begin{block}{JRoboContainer}
		Разработан JRoboContainer "--- контейнер, который позволяет избежать явного перечисления всех классов
	\end{block}
\end{frame}

\begin{frame}
